const shoppingCart = [];

// Función para agregar un producto al carrito
function addShopCar(product) {
  console.log("Estamos en page", product);

  // Agrega el producto al arreglo shoppingCart
  shoppingCart.push(product);

  console.log(`El producto: ${product.title} ha sido agregado al carrito`);
  console.log(shoppingCart); // Muestra el contenido del carrito en la consola
}

export { addShopCar };
// Componente para mostrar el carrito
export default function Page() {
  return (
    <div>
      <h2>Carrito de Compras</h2>
      <ul>
        {shoppingCart.map((product, index) => (
          <li key={index}>
            <p>Título: {product.title}</p>
            <p>Precio: ${product.price}</p>
          </li>
        ))}
      </ul>
    </div>
  );
}
